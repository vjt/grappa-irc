- [ ] **Step 1: Write the module**

```elixir
defmodule Grappa.IRC.LineSplit do
  @moduledoc """
  Splits a PRIVMSG body into wire-frame-fitting fragments.

  The IRC server's max line length (`LINELEN`, advertised via
  `005 RPL_ISUPPORT LINELEN=<N>` or RFC 2812's 512-byte default)
  bounds the WHOLE wire frame including the `PRIVMSG <target> :`
  prefix and trailing `\\r\\n`. This module computes the per-frame
  body budget, splits the body on grapheme boundaries (UTF-8
  safe), and returns one body per fragment. Caller wraps each
  in the envelope.

  ## Why server-side

  Per CLAUDE.md "IRC is bytes; the web is UTF-8" + "one parser,
  on the server" — payload framing belongs to grappa, not
  cicchetto. cic POSTs an arbitrary-length string; grappa fans
  out the fragments. Each fragment becomes its own scrollback row
  + its own upstream PRIVMSG, matching what every other IRC
  client renders + what the operator's own past view will
  reconstruct.

  ## CTCP awareness

  A body beginning with `\\x01ACTION ` and ending with `\\x01`
  is a CTCP ACTION. Fragmenting NAIVELY would emit
  `\\x01ACTION text-chunk-1` (no trailing `\\x01`) and
  `text-chunk-2\\x01` (no leading envelope) — both garbage on
  the wire. This module preserves the envelope on every
  fragment so each one is a self-contained valid CTCP message.
  Budget accounts for the per-fragment overhead.

  Other CTCP verbs (`\\x01VERSION\\x01`, `\\x01PING ...\\x01`,
  DCC, etc.) are single-line by convention; this module's CTCP
  detection only triggers for ACTION.
  """

  @doc \"""
  Splits `body` into fragments that fit within `linelen` bytes
  per wire frame, given the target prefix.

  Returns a non-empty list of UTF-8 strings, each one a valid
  PRIVMSG body for `target`. Single-fragment input returns
  `[body]` unchanged (fast path).
  \"""
  @spec split_privmsg_body(String.t(), String.t(), pos_integer()) ::
          [String.t(), ...]
  def split_privmsg_body(body, target, linelen)
      when is_binary(body) and is_binary(target) and is_integer(linelen) and linelen > 0 do
    overhead = byte_size("PRIVMSG #{target} :") + byte_size("\r\n")
    budget = linelen - overhead

    cond do
      budget <= 0 -> [body]  # malformed; caller's invariant problem
      byte_size(body) <= budget -> [body]
      ctcp_action?(body) -> split_ctcp_action(body, budget)
      true -> split_plain(body, budget)
    end
  end

  defp ctcp_action?(<<"\x01ACTION ", _::binary>> = body),
    do: String.ends_with?(body, "\x01")

  defp ctcp_action?(_), do: false

  defp split_plain(body, budget) do
    body
    |> String.graphemes()
    |> chunk_by_bytes(budget, [], [], 0)
  end

  defp split_ctcp_action(body, budget) do
    # Strip the envelope; split the inner text; re-wrap each fragment.
    inner =
      body
      |> String.replace_prefix("\x01ACTION ", "")
      |> String.replace_suffix("\x01", "")

    envelope_overhead = byte_size("\x01ACTION ") + byte_size("\x01")
    inner_budget = budget - envelope_overhead

    if inner_budget <= 0 do
      [body]
    else
      inner
      |> String.graphemes()
      |> chunk_by_bytes(inner_budget, [], [], 0)
      |> Enum.map(fn chunk -> "\x01ACTION " <> chunk <> "\x01" end)
    end
  end

  defp chunk_by_bytes([], _budget, current_chunk, acc, _current_size) do
    final =
      case current_chunk do
        [] -> Enum.reverse(acc)
        chunk -> Enum.reverse([IO.iodata_to_binary(Enum.reverse(chunk)) | acc])
      end

    case final do
      [] -> [""]  # impossible on documented path; keep return non-empty
      list -> list
    end
  end

  defp chunk_by_bytes([g | rest], budget, current_chunk, acc, current_size) do
    g_size = byte_size(g)

    cond do
      g_size > budget ->
        # Single grapheme larger than the budget (e.g. a 6-byte
        # emoji with a 5-byte budget). Emit it as its own fragment
        # — best-effort; upstream will truncate but at least the
        # grapheme is intact.
        emit_chunk_and_recurse([g | rest], budget, current_chunk, acc, current_size, g, rest)

      current_size + g_size > budget ->
        chunk_str = IO.iodata_to_binary(Enum.reverse(current_chunk))
        chunk_by_bytes(rest, budget, [g], [chunk_str | acc], g_size)

      true ->
        chunk_by_bytes(rest, budget, [g | current_chunk], acc, current_size + g_size)
    end
  end

  # Defensive: emit current chunk if non-empty, emit oversize grapheme
  # as its own fragment, continue with rest.
  defp emit_chunk_and_recurse(_args_for_doc, budget, current_chunk, acc, _current_size, g, rest) do
    acc =
      case current_chunk do
        [] -> acc
        chunk -> [IO.iodata_to_binary(Enum.reverse(chunk)) | acc]
      end

    chunk_by_bytes(rest, budget, [], [g | acc], 0)
  end
end
```

NOTE: the `emit_chunk_and_recurse/7` signature above looks like a typo (`_args_for_doc`). Adjust during implementation — pass only what's needed (probably drop unused params), Credo will complain about underscored args otherwise. The shape captures the intent: oversize grapheme = its own fragment.

defmodule Grappa.IRC.LineSplitTest do
  use ExUnit.Case, async: true
  use ExUnitProperties

  alias Grappa.IRC.LineSplit

  describe "split_privmsg_body/3 basics" do
    test "returns [body] when body fits the budget" do
      assert LineSplit.split_privmsg_body("hello", "#channel", 512) == ["hello"]
    end

    test "splits a body that exceeds the budget" do
      # PRIVMSG #c : adds 13 bytes; \r\n adds 2; budget = 80 - 15 = 65
      body = String.duplicate("a", 100)
      fragments = LineSplit.split_privmsg_body(body, "#c", 80)
      assert length(fragments) >= 2

      for fragment <- fragments do
        envelope = "PRIVMSG #c :" <> fragment <> "\r\n"
        assert byte_size(envelope) <= 80
      end

      assert Enum.join(fragments) == body
    end

    test "preserves CTCP ACTION envelope on every fragment" do
      action = "\x01ACTION " <> String.duplicate("b", 100) <> "\x01"
      fragments = LineSplit.split_privmsg_body(action, "#c", 80)
      assert length(fragments) >= 2

      for fragment <- fragments do
        assert String.starts_with?(fragment, "\x01ACTION ")
        assert String.ends_with?(fragment, "\x01")
      end
    end

    test "splits on grapheme boundaries (UTF-8 safe)" do
      # 100 emojis (each ~4 bytes), budget 80 - 15 = 65
      body = String.duplicate("🍕", 100)
      fragments = LineSplit.split_privmsg_body(body, "#c", 80)
      assert length(fragments) >= 2

      for fragment <- fragments do
        # Each fragment must round-trip as valid UTF-8
        assert String.valid?(fragment)
        envelope = "PRIVMSG #c :" <> fragment <> "\r\n"
        assert byte_size(envelope) <= 80
      end
    end

    test "single grapheme larger than budget emits it as own fragment" do
      # Edge case: budget = 1 byte, but emoji is 4 bytes.
      assert [_ | _] = LineSplit.split_privmsg_body("🍕", "#c", 16)
    end
  end

  describe "property: every fragment ≤ budget" do
    property "byte_size of each fragment + envelope ≤ linelen" do
      check all body <- string(:utf8, min_length: 1, max_length: 200),
                linelen <- integer(80..512) do
        target = "#test"
        fragments = LineSplit.split_privmsg_body(body, target, linelen)

        for fragment <- fragments do
          envelope = "PRIVMSG #{target} :" <> fragment <> "\r\n"
          # Allow 1-byte slack for the single-grapheme-oversize edge.
          assert byte_size(envelope) <= linelen + 8
        end
      end
    end
  end
end

New shape:

```elixir
defp handle_persisting_send(target, body, state) do
  fragments = Grappa.IRC.LineSplit.split_privmsg_body(body, target, state.linelen)

  case persist_and_send_fragments(target, fragments, state, nil) do
    {:ok, last_message} -> {:reply, {:ok, last_message}, state}
    {:error, _} = err -> {:reply, err, state}
  end
end

defp persist_and_send_fragments(_target, [], _state, last_message),
  do: {:ok, last_message}

defp persist_and_send_fragments(target, [fragment | rest], state, _last) do
  attrs =
    Session.put_subject_id(
      %{
        network_id: state.network_id,
        channel: target,
        server_time: System.system_time(:millisecond),
        kind: :privmsg,
        sender: state.nick,
        body: fragment,
        meta: %{},
        dm_with: Scrollback.dm_peer(:privmsg, target, state.nick, state.nick)
      },
      state.subject
    )

  with {:ok, message} <- Scrollback.persist_event(attrs),
       :ok <-
         Grappa.PubSub.broadcast_event(
           Topic.channel(state.subject_label, state.network_slug, target),
           Wire.message_payload(message)
         ),
       :ok <- Client.send_privmsg(state.client, target, fragment) do
    persist_and_send_fragments(target, rest, state, message)
  else
    {:error, _} = err -> err
  end
end
```

The `last_message` is what the caller expects (HTTP response uses the FINAL fragment's id; matches what cic's scrollback view will render as the "sent" row). Out-of-order persist→broadcast→send is preserved per the existing comment block above the original function ("the row is already in scrollback so the sender's view is consistent").

```ts
// BUGHUNT-1 B — seed the archive list on edge-trigger open. The
// mobile chip (`BottomBar.tsx`'s `.bottom-bar-archive-chip`) calls
// `setArchiveModalNetwork(slug)` to open the modal but does NOT
// call `loadArchive(slug)` — mobile operators never expand the
// Sidebar `<details>` that fires the load, so without this effect
// the modal renders "no archived windows" until the user
// archives something (which triggers an `archive_changed` event
// that re-fetches). Mirrors the load-when-opened concern with the
// mount-component-owns-state pattern: ArchiveModal seeds itself.
let lastSeededSlug: string | null = null;
createEffect(() => {
  const slug = archiveModalNetwork();
  if (slug !== null && slug !== lastSeededSlug) {
    lastSeededSlug = slug;
    void loadArchive(slug);
  } else if (slug === null) {
    lastSeededSlug = null;
  }
});
```

The `lastSeededSlug` guard prevents re-loading on every reactivity
tick — only edge-trigger open (null → slug, or slug-A → slug-B).
Re-opening the same slug AFTER close re-fires the load (refresh
semantics per `archive.ts:18-20`).
